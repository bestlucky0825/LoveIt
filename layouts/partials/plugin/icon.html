{{- $type := index . "type" | default "fa" -}} {{- $classList := split (index . "class") " " -}} {{- $context := index . "context" -}} {{- /* Fork Awesome */ -}} {{- if eq $type "fo" -}} {{- $newClassList := slice -}} {{- range $classList -}} {{- if eq . "fa" -}} {{- $newClassList = $newClassList | append "fo" -}} {{- else -}} {{- $newClassList = $newClassList | append (replaceRE `^fa\-(.+)$` `fo-$1` .) -}} {{- end -}} {{- end -}} {{- $classList = $newClassList -}} {{- $scratch := $context.Scratch.Get "scratch" -}} {{- $scratch.Set "forkawesome" true -}} {{- /* default is Font Awesome */ -}} {{- /* Others */ -}} {{- else if ne $type "fa" -}} {{- $scratch := $context.Scratch.Get "scratch" -}} {{- $scratch.Set "iconfont" true -}} {{- end -}}
